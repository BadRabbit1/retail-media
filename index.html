  const m=String(v).match(/(\d{1,2})\.(\d{1,2})\.(\d{4})/);

  if(m) return new Date(+m[3],+m[2]-1,+m[1]);

  const d=new Date(v); return isNaN(d)?null:d;

};

const parseShort = (v,yr) => {

  if(!v) return null; const s=String(v).trim();

  let m=s.match(/^(\d{1,2})[-/]([a-zа-яіє]{3,})/i);

  if(m){const mo=MONTH_MAP[m[2].toLowerCase().slice(0,3)];if(mo) return new Date(yr,mo-1,+m[1]);}

  m=s.match(/^([a-zа-яіє]{3,})[-/](\d{1,2})/i);

  if(m){const mo=MONTH_MAP[m[1].toLowerCase().slice(0,3)];if(mo) return new Date(yr,mo-1,+m[2]);}

  m=s.match(/^(\d{1,2})[./](\d{1,2})$/); if(m) return new Date(yr,+m[2]-1,+m[1]);

  return null;

};

const dayTs = d => d?new Date(d.getFullYear(),d.getMonth(),d.getDate()).getTime():null;

const normUrl = u => String(u||'').replace('/uk/','/ru/').replace('/ru/ru/','/ru/').trim().toLowerCase().replace(/\/+$/,'');

const expandTags = raw => {

  const p=String(raw||'main').split(',').map(t=>t.trim().toLowerCase()).filter(t=>t&&!IGNORE_TAG.test(t));

  return p.length?p:['main'];

};


async function sha256(s) {

  const buf=await crypto.subtle.digest('SHA-256',new TextEncoder().encode(s));

  return Array.from(new Uint8Array(buf)).map(b=>b.toString(16).padStart(2,'0')).join('');

}

function parseBanners(rows) {

  for(const r of rows){

    if(['итого','total','разом'].includes(date)) continue;

    const pur=toNum(r['Sessions with purchase']);

    const d=parseFull(r['Дата']);

      map[key].imp+=imp; map[key].clicks+=clk; map[key].purch+=pur;

      if(d) map[key].dates.add(d.getTime());

    }

  }

  return Object.values(map).filter(c=>c.imp>0).map(c=>{

    const days=c.dates.size||1;

}


// ─── paid CSV parser ─────────────────────────────────────────────

function parsePaidCSV(raw,yr) {

  const lines=raw.replace(/\r/g,'').split('\n').filter(l=>l.trim());

  if(!lines.length) return [];

  const sep=[';','\t',','].find(s=>(lines[0].match(new RegExp('\\'+s,'g'))||[]).length>=1)||';';

  const rows=lines.map(l=>l.split(sep).map(c=>c.replace(/^"|"$/g,'').trim()));

  const isUrl=v=>String(v).startsWith('http')||String(v).includes('kasta.ua');

  const isDate=v=>{const s=String(v).trim();return /^\d{1,2}[-/][a-zа-яіє]{3}/i.test(s)||/^\d{1,2}[./]\d{1,2}$/.test(s);};

  const isHeader=r=>r.every(c=>!isUrl(c)&&!isDate(c)&&String(c).length<30);

  const dataRows=rows.length>1&&isHeader(rows[0])?rows.slice(1):rows;

  const entries=[];const urlCounts={};

  for(const r of dataRows){

    const urlIdx=r.findIndex(c=>isUrl(c));if(urlIdx<0) continue;

    const brandIdx=r.findIndex((c,i)=>i!==urlIdx&&String(c).trim()&&!isDate(c)&&!isUrl(c));if(brandIdx<0) continue;

    const brand=r[brandIdx].replace(/'/g,"'").trim();

    if(!brand||/^(brand|бренд|назва)$/i.test(brand)) continue;

    const url=r[urlIdx].trim();

    const dateCols=r.map((c,i)=>({v:c,i})).filter(({v})=>isDate(v));

    let s=null,e=null;

    if(dateCols.length>=2){const d0=parseShort(dateCols[0].v,yr),d1=parseShort(dateCols[1].v,yr);if(d0&&d1){[s,e]=d0<=d1?[d0,d1]:[d1,d0];}}

    else if(dateCols.length===1){s=e=parseShort(dateCols[0].v,yr);}

    const n=normUrl(url);

    if(urlCounts[n]===undefined) urlCounts[n]=0; else urlCounts[n]++;

    entries.push({brand,url,normUrl:n,startDate:s,endDate:e,campIdx:urlCounts[n]});

  }

  return entries;

}


// ─── story parser ────────────────────────────────────────────────

function parseStories(rows,paidEntries) {

  if(!paidEntries.length) return [];

  const yrArr=rows.flatMap(r=>{const d=parseFull(r['Дата']);return d?[d.getFullYear()]:[]});

  const yr=yrArr.length?yrArr.sort()[Math.floor(yrArr.length/2)]:new Date().getFullYear();

  const urlToSlots={};

  for(const e of paidEntries){

    if(!urlToSlots[e.normUrl]) urlToSlots[e.normUrl]=[];

    urlToSlots[e.normUrl].push({brand:e.brand,campIdx:e.campIdx,

      start:e.startDate?dayTs(e.startDate):null,end:e.endDate?dayTs(e.endDate):null});

  }

  const allPaid=new Set(Object.keys(urlToSlots));

  const getUrl=r=>{

    for(const c of['ID.1','id.1']){const v=String(r[c]||'');if(v.startsWith('http')||v.includes('kasta.ua')) return v;}

    return Object.values(r).find(v=>String(v).startsWith('http'))||'';

  };

  const map={};

  for(const r of rows){

    const date=String(r['Дата']||'').toLowerCase();

    if(['итого','total','разом'].includes(date)) continue;

    const ru=getUrl(r);if(!ru||ru==='-') continue;

    const n=normUrl(ru);if(!allPaid.has(n)) continue;

    const sd=parseFull(r['Дата']);const ts=sd?dayTs(sd):null;

    const slots=urlToSlots[n];

    let slot=null;

    if(ts!==null){

      slot=slots.find(s=>s.start!==null&&s.end!==null&&ts>=s.start&&ts<=s.end);

      if(!slot){let bd=Infinity;for(const s of slots){const ref=s.end??s.start;if(ref!==null){const d=Math.abs(ts-ref);if(d<bd){bd=d;slot=s;}}}slot=slot||slots[0];}

    } else slot=slots[0];

    const key=`${n}||${slot.campIdx}`;

    if(!map[key]) map[key]={brand:slot.brand,imp:0,clicks:0,purch:0,dates:new Set()};

    map[key].imp+=toNum(r['Sessions with impressions']);

    map[key].clicks+=toNum(r['Sessions with click']);

    map[key].purch+=toNum(r['Sessions with purchase']);

    if(sd) map[key].dates.add(sd.getTime());

  }

  return Object.values(map).filter(c=>c.imp>0).map(c=>{

    const days=c.dates.size||1;

    return{title:c.brand,imp:c.imp,clicks:c.clicks,purch:c.purch,days,

      CTR:c.imp>0?c.clicks/c.imp*100:0,Conv:c.clicks>0?c.purch/c.clicks*100:0};

  });

}


// ─── benchmark builder ───────────────────────────────────────────

function buildStats(camps) {

  const by={};

  for(const c of camps){if(!by[c.instrument])by[c.instrument]=[];by[c.instrument].push(c);}

  const maxN=Math.max(1,...Object.values(by).map(g=>g.length));

  const stats=Object.entries(by).map(([inst,cs])=>{

    const ctrs=cs.map(c=>c.CTR),convs=cs.map(c=>c.Conv).filter(x=>isFinite(x));

    const imps=cs.map(c=>c.imp),days=cs.map(c=>c.days);

    const medCTR=median(ctrs),medConv=median(convs);

    const avgImp=imps.reduce((a,b)=>a+b,0)/imps.length;

    const avgIPD=(days.reduce((a,b)=>a+b,0)/days.length||1);

    const avgIPDval=avgImp/avgIPD;

    const purchPerC=(medCTR/100)*(medConv/100)*avgImp;

    const conf=Math.log1p(cs.length)/Math.log1p(maxN);

    return{instrument:inst,n:cs.length,

      medCTR:+medCTR.toFixed(3),p25CTR:+pctile(ctrs,25).toFixed(3),p75CTR:+pctile(ctrs,75).toFixed(3),

      medConv:+medConv.toFixed(2),p25Conv:+pctile(convs,25).toFixed(2),p75Conv:+pctile(convs,75).toFixed(2),

      avgImp:Math.round(avgImp),avgIPD:Math.round(avgIPDval),purchPerC:+purchPerC.toFixed(1),

      conf:+conf.toFixed(2),scoreRaw:purchPerC*conf};

  });

  const maxScore=Math.max(1,...stats.map(s=>s.scoreRaw));

  return stats.map(s=>({...s,score:+(s.scoreRaw/maxScore*100).toFixed(1)})).sort((a,b)=>b.score-a.score);

}


// ─── retention parser ────────────────────────────────────────────

function parseRetentionCSV(raw) {

  const lines=raw.replace(/\r/g,'').split('\n').filter(l=>l.trim());

  if(!lines.length) return {push:[],email:[],viber:[]};

  const hasHeader=lines[0].toLowerCase().includes('activity')||lines[0].toLowerCase().includes('media_type');

  const dataLines=hasHeader?lines.slice(1):lines;

  let lastDate='';

  const rows=dataLines.map(line=>{

    const p=line.split(';').map(s=>s.trim());

    const date=(p[0]||'').trim()||lastDate;

    if(date) lastDate=date;

    return{date,type:p[1]||'',segment:p[2]||'',

      sent:toNum(p[3]),del:toNum(p[4]),opens:toNum(p[5]),clicks:toNum(p[6])};

  }).filter(r=>r.type&&r.segment);


  const pushRows=rows.filter(r=>r.type==='mobilepush');

  const emailRows=rows.filter(r=>r.type==='email');

  const viberRows=rows.filter(r=>r.type==='viber');


  // Group push by base name

  const pushMap={};

  for(const r of pushRows){

    const isAnd=r.segment.endsWith('_and]');

    const isIos=r.segment.endsWith('_ios]');

    const base=r.segment.replace(/_and\]$/,']').replace(/_ios\]$/,']');

    if(!pushMap[base]) pushMap[base]={segment:base,dates:new Set(),and:{sent:0,del:0,opens:0,clicks:0},ios:{sent:0,del:0,opens:0,clicks:0}};

    if(r.date) pushMap[base].dates.add(r.date);

    const t=isIos?pushMap[base].ios:pushMap[base].and;

    t.sent+=r.sent;t.del+=r.del;t.opens+=r.opens;t.clicks+=r.clicks;

  }

  const push=Object.values(pushMap).map(c=>{

    const total={sent:c.and.sent+c.ios.sent,del:c.and.del+c.ios.del,opens:c.and.opens+c.ios.opens,clicks:c.and.clicks+c.ios.clicks};

    return{segment:c.segment,dates:[...c.dates].sort(),and:c.and,ios:c.ios,total,

      openRate:pct(total.opens,total.del),clickRate:pct(total.clicks,total.del),

      delRate:pct(total.del,total.sent)};

  });


  const email=emailRows.map(r=>({...r,delRate:pct(r.del,r.sent),openRate:pct(r.opens,r.del),clickRate:pct(r.clicks,r.del)}));

  const viber=viberRows.map(r=>({...r,delRate:pct(r.del,r.sent),clickRate:pct(r.clicks,r.del)}));

  return{push,email,viber};

}

        <td class="tl">${esc(seg)}</td><td class="tl" style="color:var(--text2);font-size:11px">${esc(c.date)}</td>

        <td>${fmtK(c.sent)}</td><td>${fmtK(c.del)}</td><td>${fmtK(c.opens)}</td>

        <td style="color:${c.openRate>=30?'var(--green)':''}">${c.openRate.toFixed(1)}%</td>

        <td>${fmtK(c.clicks)}</td>

        <td>${c.clickRate.toFixed(2)}%</td>

      </tr>`;

    }

    email+=`</tbody></table></div>`;

  }

  email+=`</div>`;


  // Viber

  let viber=`<div id="ret-viber" class="ret-section" style="display:none">`;

  if(!ret.viber.length){viber+=`<div class="empty" style="padding:24px">Немає даних Viber</div>`;}

  else{

    viber+=`<div class="tbl-wrap"><table class="dt">

      <thead><tr><th class="tl">Кампанія</th><th class="tl">Дата</th>

        <th>Відправлено</th><th>Доставлено</th><th>Доставк.</th>

        <th>Кліків</th><th>Click rate</th>

      </tr></thead><tbody>`;

    for(const c of ret.viber){

      const seg=c.segment.replace(/^\[/,'').replace(/\]$/,'');

      viber+=`<tr>

        <td class="tl">${esc(seg)}</td><td class="tl" style="color:var(--text2);font-size:11px">${esc(c.date)}</td>

        <td>${fmtK(c.sent)}</td><td>${fmtK(c.del)}</td>

        <td>${c.delRate.toFixed(1)}%</td>

        <td>${fmtK(c.clicks)}</td>

        <td style="color:${c.clickRate>=3?'var(--green)':''}">${c.clickRate.toFixed(2)}%</td>

      </tr>`;

    }

    viber+=`</tbody></table></div>`;

  }

  viber+=`</div>`;


  el.innerHTML=tabs+push+email+viber;

}

async function saveAll(){

  const btn=document.getElementById('save-btn');

  const token=document.getElementById('gh-token').value.trim();

  const label=document.getElementById('month-label').value.trim();

  if(!token){adminMsg('Введіть GitHub Token','error');return;}

  if(!label){adminMsg('Введіть назву місяця','error');return;}

  if(!FILES.banners&&!FILES.stories&&!FILES.retention){adminMsg('Завантажте хоча б один файл','error');return;}

  if(FILES.banners&&!FILES.stories){adminMsg('Потрібен також файл сторіз','error');return;}

  if(FILES.stories&&!FILES.banners){adminMsg('Потрібен також файл банерів','error');return;}

  if((FILES.banners||FILES.stories)&&!FILES.paid){adminMsg('Потрібен список платних сторіз (CSV)','error');return;}


  localStorage.setItem('rm_gh_token',token);

  btn.disabled=true;btn.textContent='Обробляємо…';adminMsg('Обробка файлів…','info');


  try{

    let instruments=[],bannerCamps=[],storyCamps=[],retention={push:[],email:[],viber:[]};


    if(FILES.banners&&FILES.stories&&FILES.paid){

      const[bRows,sRows,paidRaw]=await Promise.all([readXls(FILES.banners),readXls(FILES.stories),readTxt(FILES.paid)]);

      const yrArr=sRows.flatMap(r=>{const d=parseFull(r['Дата']);return d?[d.getFullYear()]:[]});

      const yr=yrArr.length?yrArr.sort()[Math.floor(yrArr.length/2)]:new Date().getFullYear();

      const paidEntries=parsePaidCSV(paidRaw,yr);

      bannerCamps=parseBanners(bRows);

      storyCamps=parseStories(sRows,paidEntries);

      instruments=buildStats(bannerCamps);

      adminMsg(`Банери: ${bannerCamps.length} кампаній, ${instruments.length} інструментів. Сторіз: ${storyCamps.length} кампаній.`,'info');

    }


    if(FILES.retention){

      const raw=await readTxt(FILES.retention);

      retention=parseRetentionCSV(raw);

      adminMsg(`Ретеншн: пуш ${retention.push.length}, email ${retention.email.length}, viber ${retention.viber.length}.`,'info');

    }


    const now=new Date().toISOString();

    const record={label,updatedAt:now,instruments,bannerCamps,storyCamps,retention};


    // Load current gist data, merge

    let gistData={};

    try{gistData=await fetchGist();}catch(e){gistData={};}

    gistData.currentRecord=record;

    gistData.label=label;

    gistData.updatedAt=now;

    // Keep history (optional)

    if(!gistData.history) gistData.history={};

    const key=label.toLowerCase().replace(/\s/g,'-');

    gistData.history[key]=record;


    adminMsg('Зберігаємо до Gist…','info');

    await saveGist(gistData,token);


    // Update live view

    DATA=record;

    renderDashboard();

    adminMsg(`✓ Збережено! Дані «${label}» доступні на сайті.`,'ok');

    clearFiles();

  }catch(e){adminMsg('Помилка: '+e.message,'error');}

  btn.disabled=false;btn.textContent='💾 Зберегти та опублікувати';

}
